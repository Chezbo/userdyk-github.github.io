```python
import numpy as np
import sympy
sympy.init_printing()
from sympy import I, pi, oo

x = sympy.Symbol("x")
y = sympy.Symbol("y", real=True) 
y.is_real
```

```python
x.is_real is None
```

```python
x = sympy.Symbol("x")

sympy.sqrt(x ** 2) 
```


```python
sympy.cos(n3 * pi)
```
